#include <string>
#include <iostream>
#include <unordered_map>
#include <vector>
using namespace std;

int main()
{

	int n;
	cin >> n;
	unordered_map<string, int> m;

	for (int i = 0; i < n; ++i) {
		string s;
		cin >> s;
		m[s] = i;
	}

	vector<string> v(n);
	int cnt = 0;
	for (int i = 0; i < n; ++i) {
		cin >> v[i];
	}

	for (int i = 0; i < n; ++i) {
		for (int j = i + 1; j < n; ++j) {
			if (m[v[i]] > m[v[j]]) {
				cnt += 1;
				break;
			}
		}
	}
	cout << cnt << "\n";
	return 0;
}
