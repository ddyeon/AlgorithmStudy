```c++
#include <iostream>
#include <string>
#include <string.h>
#include <algorithm>
using namespace std;

int main() {
    string a,b;

    cin >> a >> b;

    int min_len = 100000;

    for(int i = 0; i < b.size()-a.size()+1; i++) {
        int cnt = 0;
        for(int j = 0; j < a.size(); j++)
            if(a[j] != b[i+j]) cnt++;
        min_len = min(min_len,cnt);
    }
    
    cout << min_len << endl;
}
```
