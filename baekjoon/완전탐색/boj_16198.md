#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
int n;
vector <int> v;
int max_w;
int tmp_w;

void solve() {
    if(v.size() == 2) {
        if(max_w < tmp_w) {
            max_w = tmp_w;
            return;
        }
    }
    
    for (int i = 1; i < v.size() - 1; i++) {
        int cur = v.at(i);
        tmp_w += v.at(i-1) * v.at(i+1);
        v.erase(v.begin()+i);
        solve();
        v.insert(v.begin()+i,cur);
        tmp_w = tmp_w - v.at(i-1)*v.at(i+1);
    }
}


int main() {
    cin >> n;
    
    for(int i = 0; i < n; i++) {
        int x;
        cin >> x;
        v.push_back(x);
    }
    
    solve();
    
    cout << max_w << endl;
}
